def construct_assume_role(
        environment: str,
        dev_account_id: str,
        uat_account_id: str,
        prod_account_id: str,
        auto_account_id: str,
        assume_role_name: str
):
    """Construct an IAM Role string.

    Args:
        environment: The target environment identifier ('dev', 'prod', 'uat')
        dev_account_id: The account id of TCA dev
        uat_account_id: The account id of TCA uat
        prod_account_id: The account id of TCA pro
        auto_account_id: The account id of TCA auto
        assume_role_name: The IAM Role name to be assumed by Lambda

    Returns:
        role: IAM Role arn for assume_role_name in sts_assume_role()
        account_id: Account ID for account in sts_assume_role()

    """
    if environment == "dev":
        role = "arn:aws:iam::{}:role/{}".format(dev_account_id, assume_role_name)
        return role, dev_account_id
    if environment == "uat":
        role = "arn:aws:iam::{}:role/{}".format(uat_account_id, assume_role_name)
        return role, uat_account_id
    if environment == "prod":
        role = "arn:aws:iam::{}:role/{}".format(prod_account_id, assume_role_name)
        return role, prod_account_id
    if environment == "auto":
        role = "arn:aws:iam::{}:role/{}".format(auto_account_id, assume_role_name)
        return role, auto_account_id


def sts_assume_role(
        assume_role_name: str,
        session_name: str,
        account: str,
):
    """Assume a IAM role in this or a foreign account.

    Args:
        assume_role_name: The name of the IAM role
        session_name: An arbitrary session name
        account: The account name or id. Just for logging purposes

    Returns:
        access_key: A temporary access key provided by Security Token Service (STS)
        secret_key: A temporary secret key key provided by Security Token Service (STS)
        session_token: A temporary session token provided by Security Token Service (STS)

    """
    try:
        sts_connection = boto3.client('sts')
        sts_response = sts_connection.assume_role(
            RoleArn=assume_role_name,
            RoleSessionName=session_name,
        )
        access_key = sts_response['Credentials']['AccessKeyId']
        secret_key = sts_response['Credentials']['SecretAccessKey']
        session_token = sts_response['Credentials']['SessionToken']
    except ClientError as e:
        logger.error("Error while assuming role in other account {} ... {}".format(account, e))
        raise e
    return access_key, secret_key, session_token,


def create_bucket_policy(
        service_role_arn: str,
        s3_api_calls: list,
        bucket_name: str,
        access_key: str,
        secret_key: str,
        session_token: str,
):
    """Create and put a S3 Bucket Policy.

    Args:
        service_role_arn: The IAM Role to allow communication to the specified Bucket
        s3_api_calls: The range of allowed S3 API calls
        bucket_name: The S3 Bucket name
        access_key: A temporary access key provided by Security Token Service (STS)
        secret_key: A temporary secret key key provided by Security Token Service (STS)
        session_token: A temporary session token provided by Security Token Service (STS)

    Returns:
        No returns

    """
    policy = {
        "Version": "2012-10-17",
        "Statement": [
            {
                "Sid": "AddS3Permissions",
                "Action": s3_api_calls,
                "Effect": "Allow",
                "Resource": [
                    "arn:aws:s3:::{}".format(bucket_name),
                    "arn:aws:s3:::{}/*".format(bucket_name)
                ],
                "Principal": {
                    "AWS": [
                        "{}".format(service_role_arn)
                    ]
                }
            }
        ]
    }
    # Convert the policy from JSON dict to string
    bucket_policy = json.dumps(policy)
    try:
        s3 = boto3.client(
            "s3",
            aws_access_key_id=access_key,
            aws_secret_access_key=secret_key,
            aws_session_token=session_token
        )
        s3.put_bucket_policy(
            Bucket=bucket_name,
            Policy=bucket_policy
        )
        logger.info("Put successfully bucket policy ...")
    except ClientError as e:
        if e.response['Error']['Code'] == 'NoSuchBucket':
            logger.error("The bucket you specified does not exists ... {}".format(e))
            raise e
        if e.response['Error']['Code'] == 'AccessDenied':
            logger.error("You are not allowed to perform this action ... {}".format(e))
            raise e
        else:
            logger.error("Error while putting Bucket Policy... {}".format(e))
            raise e

def put_service_policy(
        policy_name: str,
        service_role_name: str,
        s3_api_calls: list,
        bucket_name: str
):
    """Put an additional IAM Role Policy to a preexisting IAM Service Role, e.g. Sagemaker.

    Args:
        policy_name: The name of the additional IAM Role Policy
        service_role_name: The name of the preexisting IAM Service Role
        s3_api_calls: The range of allowed S3 API calls
        bucket_name: The S3 Bucket name

    Returns:
        No returns

    """
    policy = {
        "Version": "2012-10-17",
        "Statement": [{
            "Sid": "AddPermissions",
            "Effect": "Allow",
            "Action": s3_api_calls,
            "Resource": [
                "arn:aws:s3:::{}".format(bucket_name),
                "arn:aws:s3:::{}/*".format(bucket_name)
            ]
        }]
    }
    service_policy = json.dumps(policy)
    try:
        iam_client = boto3.client("iam")
        iam_client.put_role_policy(
            PolicyDocument=service_policy,
            PolicyName=policy_name,
            RoleName=service_role_name
        )
    except ClientError as e:
        if e.response['Error']['Code'] == 'NoSuchEntity':
            logger.error("The role you specified does not exists ... {}".format(e))
            raise e
        if e.response['Error']['Code'] == 'AccessDenied':
            logger.error("You are not allowed to perform this action ... {}".format(e))
            raise e
        else:
            logger.error("Error while putting service role ... {}".format(e))
            raise e

    except ClientError as e:
        if e.response['Error']['Code'] == 'NoSuchEntity':
            logger.error("The role you specified does not exists ... {}".format(e))
            raise e
        if e.response['Error']['Code'] == 'AccessDenied':
            logger.error("You are not allowed to perform this action ... {}".format(e))
            raise e
        else:
            logger.error("Error while putting service role ... {}".format(e))
            raise e


def main(event, context):
    """Orchestrate the other functions. Runs during a CloudFormation deployment as CustomResource.

    Args:
        event: The Lambda event object
        context: The Lambda context object

    Returns:
        response_data['Message']: Returns a CloudFormation signal via cfnresponse.send().
        The return determines a Success or Failure for this CustomResource

    """
    logger.info("Starting cross-account bucket policy change ... ")
    logger.info(event)

    service_role = event["ResourceProperties"]["SERVICE_ROLE"]
    target_account = event["ResourceProperties"]["TARGET_ACCOUNT"]
    target_bucket = event["ResourceProperties"]["TARGET_BUCKET"]
    actions = event["ResourceProperties"]["ACTIONS"]
    policy_name = event["ResourceProperties"]["IAM_POLICY_NAME"]
    env = event["ResourceProperties"]["ENVIRONMENT"]

    if event['RequestType'] == 'Create':
        logger.info("Allow communication between {} and IAM service role {}  ".format(target_bucket, service_role))
        try:
            role, account = construct_assume_role(
                dev_account_id=dev_account,
                uat_account_id=uat_account,
                prod_account_id=prod_account,
                auto_account_id=auto_account,
                environment=env,
                assume_role_name=role_name
            )
            logger.info("Trying to assume IAM role {} in {} ...".format(role, env))
            acc_key, sec_key, sess_token = sts_assume_role(
                session_name="AdoptBucketPolicy",
                account=target_account,
                assume_role_name=role,
            )
            logger.info("Trying to put bucket policy on bucket {} in account {} ...".format(target_bucket, account))
            create_bucket_policy(
                service_role_arn=service_role,
                s3_api_calls=actions,
                bucket_name=target_bucket,
                access_key=acc_key,
                secret_key=sec_key,
                session_token=sess_token,
            )
            logger.info("Trying to put role policy on {} ...".format(service_role))
            put_service_policy(
                bucket_name=target_bucket,
                s3_api_calls=actions,
                service_role_name=service_role.split("/")[1],
                policy_name=policy_name
            )
            response_data['Message'] = "Success"
            cfnresponse.send(event, context, cfnresponse.SUCCESS, response_data)
        except Exception as e:
            logger.error(
                """Error while allowing communication between {} and IAM service role {} ... {}
                """.format(target_bucket, service_role, e))
            response_data['Message'] = "Failure"
            cfnresponse.send(event, context, cfnresponse.FAILED, response_data)
    elif event['RequestType'] == 'Delete':
        logger.info("Remove communication between {} and IAM service role {}  ".format(target_bucket, service_role))
        try:
            role, account = construct_assume_role(
                dev_account_id=dev_account,
                uat_account_id=uat_account,
                prod_account_id=prod_account,
                auto_account_id=auto_account,
                environment=env,
                assume_role_name=role_name
            )
            acc_key, sec_key, sess_token = sts_assume_role(
                assume_role_name=role,
                session_name="AdoptBucketPolicy",
                account=target_account,
            )
            logger.info("Deleting bucket policy on bucket {} in account {} ...".format(target_bucket, account))
            delete_bucket_policy(
                bucket_name=target_bucket,
                access_key=acc_key,
                secret_key=sec_key,
                session_token=sess_token,
            )
            logger.info("Deleting service role {} in this account {} ... ".format(service_role.split("/")[1], account))
            delete_service_policy(
                service_role_name=service_role.split("/")[1],
                policy_name=policy_name
            )
            response_data['Message'] = "Success"
            cfnresponse.send(event, context, cfnresponse.SUCCESS, response_data)
        except Exception as e:
            logger.error(
                """
                Error while removing communication between {} and IAM service role {} ... {}
                """.format(target_bucket, service_role, e))
            response_data['Message'] = "Failure"
            cfnresponse.send(event, context, cfnresponse.FAILED, response_data)

    """Put an additional IAM Role Policy to a preexisting IAM Service Role, e.g. Sagemaker.

    Args:
        policy_name: The name of the additional IAM Role Policy
        service_role_name: The name of the preexisting IAM Service Role
        s3_api_calls: The range of allowed S3 API calls
        bucket_name: The S3 Bucket name

    Returns:
        No returns

    """
    policy = {
        "Version": "2012-10-17",
        "Statement": [{
            "Sid": "AddPermissions",
            "Effect": "Allow",
            "Action": s3_api_calls,
            "Resource": [
                "arn:aws:s3:::{}".format(bucket_name),
                "arn:aws:s3:::{}/*".format(bucket_name)
            ]
        }]
    }
    service_policy = json.dumps(policy)
    try:
        iam_client = boto3.client("iam")
        iam_client.put_role_policy(
            PolicyDocument=service_policy,
            PolicyName=policy_name,
            RoleName=service_role_name
        )
    except ClientError as e:
        if e.response['Error']['Code'] == 'NoSuchEntity':
            logger.error("The role you specified does not exists ... {}".format(e))
            raise e
        if e.response['Error']['Code'] == 'AccessDenied':
            logger.error("You are not allowed to perform this action ... {}".format(e))
            raise e
        else:
            logger.error("Error while putting service role ... {}".format(e))
            raise e

def sts_assume_role(
        assume_role_name: str,
        session_name: str,
        account: str,
):
    """Assume a IAM role in this or a foreign account.

    Args:
        assume_role_name: The name of the IAM role
        session_name: An arbitrary session name
        account: The account name or id. Just for logging purposes

    Returns:
        access_key: A temporary access key provided by Security Token Service (STS)
        secret_key: A temporary secret key key provided by Security Token Service (STS)
        session_token: A temporary session token provided by Security Token Service (STS)

    """
    try:
        sts_connection = boto3.client('sts')
        sts_response = sts_connection.assume_role(
            RoleArn=assume_role_name,
            RoleSessionName=session_name,
        )
        access_key = sts_response['Credentials']['AccessKeyId']
        secret_key = sts_response['Credentials']['SecretAccessKey']
        session_token = sts_response['Credentials']['SessionToken']
    except ClientError as e:
        logger.error("Error while assuming role in other account {} ... {}".format(account, e))
        raise e
    return access_key, secret_key, session_token,
